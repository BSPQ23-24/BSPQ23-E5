186 private void loadRoutesData(DefaultTableModel model) {
187 List<TicketDTO> tickets = TicketController.getInstance().getTicketByUser(user.getEmail());
188 Set<RouteDTO> uniqueRoutes = new HashSet<>();
189 for (TicketDTO ticket : tickets) {
190 RouteDTO route = ticket.getSchedule().getRoute();
191 uniqueRoutes.add(route);
192 }
193 for (RouteDTO route : uniqueRoutes) {
194 Object[] rowData = {route.getName(), route.getStartPoint(), route.getEndPoint(), route.getTotalDistance()};
195 model.addRow(rowData);
196 }
197 }
