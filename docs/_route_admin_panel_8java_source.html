118 private void updateTableModel(List<RouteDTO> routes) {
119 tableModel.setRowCount(0);
120 if (routes != null) {
121 for (RouteDTO route : routes) {
122 String stations = route.getStations().stream()
123 .map(station -> station.getLocation())
124 .collect(Collectors.joining(", "));
125 String buses = route.getBuses().stream()
126 .map(bus -> bus.getLicensePlate())
127 .collect(Collectors.joining(", "));
128 tableModel.addRow(new Object[]{route.getName(), route.getStartPoint(), route.getEndPoint(), route.getTotalDistance() + " km", stations, buses});
129 }
130 }
