121 private JFreeChart createPieChart() {
122 DefaultPieDataset<String> dataset = new DefaultPieDataset<>();
123
124 List<RouteDTO> routes = RouteController.getInstance().getAllRoutes();
125 for (RouteDTO route : routes) {
126 int totalPassengers = route.getTotalPassengers();
127 dataset.setValue(route.getName(), totalPassengers);
128 }
129
130 JFreeChart pieChart = ChartFactory.createPieChart(
131 messages.getString("pieChartTitle"),
132 dataset,
133 true, true, false);
134
135 @SuppressWarnings("unchecked")
136 PiePlot<String> plot = (PiePlot<String>) pieChart.getPlot();
137 plot.setBackgroundPaint(colorBackground);
138 pieChart.setBackgroundPaint(colorBackground);
139 pieChart.getTitle().setPaint(colorPrimary);
140
141 int colorIndex = 0;
142 Color[] pieColors = {colorPrimary, colorSecondary, colorTertiary};
143 for (String key : dataset.getKeys()) {
144 plot.setSectionPaint(key, pieColors[colorIndex % pieColors.length]);
145 colorIndex++;
146 }
147
148 plot.setLabelBackgroundPaint(colorTertiary);
149 plot.setLabelPaint(Color.BLACK);
150
151 return pieChart;
152 }

154 private JTable createTopRoutesTable() {
155 String[] columns = messages.getString("tableColumnsHome").split(",");
156 DefaultTableModel model = new DefaultTableModel(columns, 0) {
157 @Override
158 public boolean isCellEditable(int row, int column) {
159 return false;

162

164
165 for (int i = 0; i < routes.size(); i++) {

168 model.addRow(new Object[]{i + 1, route.getName(), totalPassengers});

171 return TableUtil.createConfiguredTable(model, colorBackground, colorSecondary, Color.WHITE);

174 private JLabel loadImage(String imagePath, int width, int height) {
